#+TITLE: My .bash_alias

* =ls= aliases
#+begin_src shell
alias ll='ls -alF'
alias la='ls -A'
alias l='ls -CF'
#+end_src
* =alert= for long running commands
#+begin_src shell
# e.g. sleep 10; alert
alias alert='notify-send --urgency=low -i "$([ $? = 0 ] && echo terminal || echo error)" "$(history|tail -n1|sed -e '\''s/^\s*[0-9]\+\s*//;s/[;&|]\s*alert$//'\'')"'
#+end_src
* Reload bash config
#+begin_src shell
alias reload=". ~/.bash_profile"
#+end_src
